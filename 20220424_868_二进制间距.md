class Solution {
public:
    int binaryGap(int n) {
        int pre=-1, cnt=0, res=0;

        while (n>0) {
            if (n%2==1) {
                if (pre!=-1) {
                    res=max(res,cnt-pre);
                }
                pre=cnt;
            }
            cnt++;
            n>>=1;
        }

        return res;
    }
};
