class Solution {
public:
    int maxDepth(string s) {
        int res=0, cur=0;
        for (char& ch: s) {
            if (ch=='(') {
                cur++;
                if (cur>res) {
                    res = cur;
                }
            } else if (ch==')') {
                cur--;
            }
        }
        return res;
    }
};
