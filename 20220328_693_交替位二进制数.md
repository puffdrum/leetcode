class Solution {
public:
    bool hasAlternatingBits(int n) {
        int prev=n%2, cur;
        n>>=1;
        while (n) {
            cur=n%2;
            if (cur==prev) {
                return false;
            }
            prev=cur;
            n>>=1;
        }
        return true;
    }
};
