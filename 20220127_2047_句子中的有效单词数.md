class Solution {
private:
    bool isToken(string str) {
        int n=str.size();
        int cnt1=0, cnt2=0;
        for (char& ch: str) {
            if (ch=='-') {
                cnt1++;
            } else if (ch=='!'||ch==','||ch=='.') {
                cnt2++;
            } else if (ch>='0'&&ch<='9') {
                return false;
            }
        }
        if (cnt1==0 && cnt2==0) {
            return true;
        }
        if (cnt1>1 || cnt2>1) {
            return false;
        }
        if (str[n-1]=='!' || str[n-1]==',' || str[n-1]=='.') {
            if (str[n-2]=='-') {
                return false;
            }
        } else if (str[n-1]=='-' || cnt2==1) {
            return false;
        }
        return true;
    }
public:
    int countValidWords(string sentence) {
        string str="";
        int cnt=0, cnt1;
        int i=0, j, n=sentence.size();
        while (i<n) {
            while (i<n && sentence[i]==' ') {
                i++;
            }
            if (i==n) {
                break;
            }

            if (sentence[i]=='!'||sentence[i]==','||sentence[i]=='.') {
                if ((i+1<n && sentence[i+1]==' ') || (i+1==n)) {
                    cnt++;
                    i++;
                } else {
                    while (i<n && sentence[i]!=' ') {
                        i++;
                    }
                }
                continue;
            }

            if (sentence[i]=='-') {
                while (i<n && sentence[i]!=' ') {
                    i++;
                }
                continue;
            }

            j = i;
            while (j<n && sentence[j]!=' ') {
                j++;
            }
            str = sentence.substr(i,j-i);
            if (isToken(str)) {
                cnt++;
            }
            i = j;
        }
        return cnt;
    }
};
