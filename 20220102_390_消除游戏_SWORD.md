class Solution {
public:
    int lastRemaining(int n) {
        int a1=1, an=n;
        int cnt=n, step=1;
        int k=0;
        while (cnt>1) {
            if (k%2==0) {
                a1 += step;
                if (cnt%2!=0) an -= step;
            }
            else {
                an -= step;
                if (cnt%2!=0) a1 += step;
            }
            step <<= 1;
            cnt >>= 1;
            k++;
        }
        return a1;
    }
};
