class Solution {
public:
    string convertToBase7(int num) {
        if (num==0) {
            return "0";
        }
        int x=abs(num);
        string res="";
        while (x>0) {
            res = to_string(x%7) + res;
            x /= 7;
        }
        if (num<0) {
            res = "-"+res;
        }

        return res;
    }
};
